Basic usage

Super simple example how to backup MySQL database and send it to Google Drive in compressed format tar.bz2...

                                  use Dzasa\MaratusPhpBackup\MaratusBackup;

$backup = new MaratusBackup();

//Define config for MySQL database
$dbConfig2 = array(
   'type' => 'mysql',
   'host' => "localhost",
   'port' => 3306,
   'user' => 'root',
   'pass' => 'root',
   'database' => 'db1' //if left empty, all database will be backed up
);

//Add config into database config queue
$backup->addDatabase($dbConfig2);

//Define Google Drive config for storing backup files
$gDriveConfig = array(
   'type' => 'gdrive',
   'client_id' => '',
   'client_secret' => '',
   'token_file' => 'gdrive-token.json', //file name where we will save token and other stuff for future use
   'auth_code' => ''
);

//Add config into storage queue
$backup->addStorage($gDriveConfig);

//Set archive type
$backup->backup("tar.bz2");

//Get result info from all database backups
$databasesResult = $backup->getDatabaseBackupResult();

//Get result info from all storages
$backupStorageResult = $backup->getStorageBackupResult();
